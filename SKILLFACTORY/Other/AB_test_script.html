def optimize_users(df, users_col, goal_col):
    new_rows = []
    for index, row in df.iterrows():
        for i in range(row[users_col]):
            if i < row[goal_col]:
                new_row = row.copy()
                new_row[users_col] = 1
                new_row[goal_col] = 1
                new_rows.append(new_row)
            else:
                new_row = row.copy()
                new_row[users_col] = 1
                new_row[goal_col] = 0
                new_rows.append(new_row)
    new_df = pd.DataFrame(new_rows, )
    return new_df.reset_index(drop=True)

def z_mean_conf_interval(n, x_mean, x_std, gamma=0.95):
    alpha = 1 - gamma #  
    z_crit = -norm.ppf(alpha/2) # z 
    eps = z_crit * x_std / n**0.5 # 
    lower_bound = x_mean - eps #  () 
    upper_bound = x_mean + eps #  () 
    return lower_bound, upper_bound

def diff_proportions_conf_interval(n, x_p, gamma=0.95):
    alpha = 1 - gamma #  
    diff = x_p[1] - x_p[0] #     B  A
    z_crit = -norm.ppf(alpha/2) # z 
    eps = z_crit * (x_p[0] * (1 - x_p[0])/n[0] + x_p[1] * (1 - x_p[1])/n[1]) ** 0.5 # 
    lower_bound = diff - eps #  () 
    upper_bound = diff + eps #  () 
    #      
    return round(lower_bound *100, 2), round(upper_bound * 100, 2)
